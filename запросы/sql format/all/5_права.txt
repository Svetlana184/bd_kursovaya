-- Создание схемы для безопасности
CREATE SCHEMA security;
GO

-- Функция фильтрации (пользователь видит только свои данные)
CREATE FUNCTION security.fn_UserAccessPredicate(@UserId INT)
RETURNS TABLE
WITH SCHEMABINDING
AS
RETURN
SELECT 1 AS AccessResult
WHERE 
    -- Администратор видит всё
    IS_ROLEMEMBER('DBA_Role') = 1 
    OR
    -- Приложение видит данные только текущего пользователя
    @UserId = CAST(SESSION_CONTEXT(N'UserId') AS INT);
GO

-- Применение политики к таблицам
CREATE SECURITY POLICY security.UserDataPolicy
ADD FILTER PREDICATE security.fn_UserAccessPredicate(IdUser) ON dbo.Users,
ADD FILTER PREDICATE security.fn_UserAccessPredicate(IdUser) ON dbo.Notes,
ADD FILTER PREDICATE security.fn_UserAccessPredicate(IdUser) ON dbo.Habits,
ADD FILTER PREDICATE security.fn_UserAccessPredicate(IdUser) ON dbo.Goals,
ADD FILTER PREDICATE security.fn_UserAccessPredicate(IdUser) ON dbo.Trackers,
ADD FILTER PREDICATE security.fn_UserAccessPredicate(IdUser) ON dbo.Tasks,
ADD FILTER PREDICATE security.fn_UserAccessPredicate(IdUser) ON dbo.TodoLists;
GO
